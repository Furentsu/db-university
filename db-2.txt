1 - SELECT count(*) AS `students`, YEAR(`enrolment_date`) AS `year`
    FROM `students`
    GROUP BY `year`;

2 - SELECT count(*) AS `teachers`, `office_address` AS `building`
    FROM `teachers`
    GROUP BY `building`;

3 - SELECT `exam_id`, AVG(`vote`) AS `average`
    FROM `exam_student`
    GROUP BY `exam_id`;

4 - SELECT `department_id`, COUNT(*) AS `degrees`
    FROM `degrees`
    GROUP BY `department_id`;
